# PROTACable Environment Setup

To set up the `PROTACable` environment, follow these steps in order:

conda create -n PROTACable python=3.8

conda create --name PROTACable python=3.8

conda install --no-channel-priority pytorch pytorch-gpu torchvision pytorch-lightning cudatoolkit=11.2 -c pytorch -c conda-forge -y

conda install --no-channel-priority -c schrodinger pymol -y

conda install --no-channel-priority -c conda-forge ambertools=21.9 -y

conda install --no-channel-priority -c "conda-forge/label/cf202003" openbabel -y

conda install --no-channel-priority -c conda-forge pdbfixer -y

conda install --no-channel-priority -c conda-forge gcc=9.4.0 -y

conda install --no-channel-priority -c conda-forge gxx -y

export LD_LIBRARY_PATH=/home/${USER}/.conda/envs/PROTACable/lib/:$LD_LIBRARY_PATH

pip install torch-scatter torch-sparse torch-cluster torch-spline-conv torch-geometric -f https://data.pyg.org/whl/torch-1.11+cu112.html
pip install dgl-cu111
pip install joblib
pip install git+https://github.com/bp-kelley/descriptastorus
pip install biopython
pip install matplotlib
pip install biopandas
pip install cython
```
